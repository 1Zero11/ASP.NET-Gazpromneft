    SELECT CustomerID, CustomerName, столбец_с_дубликатами 
           FROM Customers
           WHERE Country IN (SELECT Country
                FROM Customers
                GROUP BY Country
                HAVING COUNT(*) > 1)
           ORDER BY Country, CustomerID;
